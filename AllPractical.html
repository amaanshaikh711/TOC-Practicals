#practical1
#include <iostream>
using namespace std;

int main(){
    string str;
    int q0 = 0, q1 = 1, q2 = 2, q3 = 3;
    int current_state = q0;
    char ch;
    cout << "Enter the string\n" << endl;
    cin >> str;
    for (int i = 0; i < str.length(); ++i){
        ch = str[i];
        cout << "\n" << ch << " State " << current_state << endl;
        if(current_state == q0 && ch == '1'){
            current_state = q1;
        }
        else if(current_state == q1 && ch == '0'){
            current_state = q0;
        }
        else if(current_state == q1 && ch == '1'){
            current_state = q2;
        }
        else if(current_state == q2 && ch == '0'){
            current_state = q0;
        }
        else if(current_state == q2 && ch == '1'){
            current_state = q3;
        }
        else if(current_state == q3 && (ch == '0' || ch == '1')){
            current_state = q3;
        }
    }
    if(current_state == q3){
        cout << "\n" << str << " is acceptable" << endl;
    }
    else{
         cout << "\n" << str << " is not acceptable" << endl;
    }
    return 0;
}

#include <iostream>
#include <algorithm>
using namespace std;

bool is_one(char c){
    return c == '1';
}

bool is_zero(char c){
    return c == '0';
}

int main(){
    string str;
    int q0 = 0, q1 = 1;
    int current_state = q0;
    string output = "";
    char ch;
    cout << "\nEnter the string :" << endl;
    cin >> str;
    
    for(int i = 0; i < str.length(); ++i){
        ch = str[i];
        if(current_state == q0 && ch == '0'){
            current_state = q0;
            output += ch;
        }
        else if(current_state == q0 && ch == '1'){
            current_state = q1;
            output += ch;
        }
        else if(current_state == q1 && ch == '0'){
            current_state = q0;
            output += ch;
        }
        else if(current_state == q1 && ch == '1'){
            current_state = q1;
            output += ch;
        }
    }
    cout << "\nNumber of zeros : " << count_if(output.begin(), output.end(), is_zero) << endl;
    cout << "Number of ones : " << count_if(output.begin(), output.end(), is_one) << endl;
    return 0;
}

#include <iostream>
#include <stack>
using namespace std;

int main(){
    string str;
    char start_element = 'z';
    int q0 = 0, q1 = 1;
    int current_state = q0;
    stack stk;
    stk.push(start_element);
    char ch;
    cout << "\nEnter the string :" << endl;
    cin >> str;
    for (int i = 0; i < str.length(); ++i){
        ch = str[i];
        if(stk.top() != start_element){    
            if(stk.top() == ch){
                stk.push(ch);
            }
            else{
                stk.pop();
            }
        }
        else{
            stk.push(ch);
        }
    }
    if (stk.top() == start_element){
        current_state = q1;
    }
    if(current_state == q1){
        cout << "\n" << str << " is acceptable" << endl;
        cout << stk.top() << endl;
    }
    else{
        cout << "\n" << str << " is not acceptable!!" << endl;
        cout << stk.top() << endl;
    }
    return 0;
}

#practical 7
#include <iostream>
#include <stack>
using namespace std;

int main(){
    string str;
    bool flag = false;
    const char start_element = 'z';
    const char reverser = 'C';
    int q0 = 0, q1 = 1;
    int current_state = q0;
    stack stk;
    stk.push(start_element);
    char ch;
    cout << "\nEnter the string " << endl;
    cin >> str;
    
    for(int i = 0; i < str.length(); ++i){
        ch = str[i];
        if(ch == reverser){
            flag = true;
            continue;
        }
        if(!flag){
            stk.push(ch);
        }
        else{
            if(ch == stk.top()){
                stk.pop();
            }
            else{
                break;
            }
        }
    }
    if(stk.top() == start_element){
        current_state = q1;
    }
    if(current_state == q1){
        cout << "\n" << str << " is a palindrome" << endl;
    }
    else{
        cout << "\n" << str << " is not a palindrome!!" << endl;
    }
    return 0;
}

#practical 8
#include <iostream>
#include <sstream>
#include <vector> 
using namespace std;

vector tokenize(const string &input_string) {
    vector tokens;
    stringstream ss(input_string);
    string token;
    // Extract words using space as a delimiter
    while (ss >> token) {
        tokens.push_back(token);
    }
    return tokens;
}

int main() {
    string input_string;
    // Taking input from the user
    cout << "Enter a string to tokenize: ";
    getline(cin, input_string);
    // Tokenizing the input string
    vector tokens = tokenize(input_string);
    // Displaying the tokens
    cout << "Tokens: ";
    for (const string &token : tokens) {
        cout << token << " ";
    }
    return 0;
}

#practical 9
#include <iostream>
#include <vector>
#include <string>
#include <unordered_map>
#include <sstream>
using namespace std;

// A struct to represent a production rule.
struct Production {
    char non_terminal;
    vector rules; // list of productions for the non-terminal
};

// A function to generate the regular expression from the given grammar.
string generateRegex(const unordered_map> &grammar, char startSymbol) {
    stringstream regex;
    unordered_map regex_map;
    // Handle the start symbol first
    regex_map[startSymbol] = "";
    // Iterate through the grammar and build the regular expressions
    for (const auto &entry : grammar) {
        char non_terminal = entry.first;
        const vector &rules = entry.second;
        string rule_expr;
        for (const string &rule : rules) {
            if (rule == "ε") {
                // Handle epsilon production
                rule_expr = rule_expr.empty() ? "" : rule_expr + "|";
                rule_expr += "ε";
            } else {
                // Handle terminal and non-terminal production
                for (char symbol : rule) {
                    rule_expr += symbol;
                }
            }
        }
        regex_map[non_terminal] = rule_expr;
    }
    // Build the final regular expression string from the start symbol
    regex << regex_map[startSymbol];
    return regex.str();
}

int main() {
    // Example Grammar
    // A → aB | ε
    // B → bA | c
    unordered_map> grammar;
    grammar['A'] = {"aB", "ε"};
    grammar['B'] = {"bA", "c"};
    char startSymbol = 'A';
    // Generate regular expression from the grammar
    string regex = generateRegex(grammar, startSymbol);
    cout << "Generated Regular Expression: " << regex << endl;
    return 0;
}
